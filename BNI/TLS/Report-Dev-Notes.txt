SELECT 
    "TLS_TRUCK_LOG_BNI"."DAILY_ROW_NUM", "TLS_TRUCK_LOG_BNI"."TIME_IN", "TLS_TRUCK_LOG_BNI"."TIME_OUT", "TLS_TRUCK_LOG_BNI"."CHECKED_IN_BY", "TLS_TRUCK_LOG_BNI"."CHECKED_OUT_BY", "TLS_TRUCK_LOG_BNI"."TRUCKING_COMPANY", "TLS_TRUCK_LOG_BNI"."DRIVER_NAME", "TLS_TRUCK_LOG_BNI"."COMMODITY_CODE", "TLS_TRUCK_LOG_BNI"."CUSTOMER_CODE", "TLS_TRUCK_LOG_BNI"."LR_NUM", "TLS_TRUCK_LOG_BNI"."BOL", "TLS_TRUCK_LOG_BNI"."SEAL_NUM", "TLS_TRUCK_LOG_BNI"."WAREHOUSE",
    ("TLS_TRUCK_LOG_BNI"."TIME_OUT"-"TLS_TRUCK_LOG_BNI"."TIME_IN")*24*60
FROM
    "SAG_OWNER"."TLS_TRUCK_LOG" "TLS_TRUCK_LOG_BNI"
WHERE
    TLS_TRUCK_LOG_BNI."TIME_IN" >= TO_DATE ('05-09-2006 00:00:00', 'DD-MM-YYYY HH24:MI:SS') AND
    TLS_TRUCK_LOG_BNI."TIME_IN" <= TO_DATE ('05-09-2006 23:59:59', 'DD-MM-YYYY HH24:MI:SS')
ORDER BY
    "TLS_TRUCK_LOG_BNI"."DAILY_ROW_NUM"
